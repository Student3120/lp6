# Load built-in dataset
data(iris)

# Remove Species column (for unsupervised learning)
newiris <- iris
newiris$Species <- NULL

# Apply K-means clustering (3 clusters)
kc <- kmeans(newiris, 3)

# Print clustering result
print(kc)

# Compare actual vs predicted clusters
print(table(iris$Species, kc$cluster))

# Plot clusters
plot(newiris$Sepal.Length, newiris$Sepal.Width,
     col = kc$cluster,
     main = "K-Means Clustering on Iris Dataset",
     xlab = "Sepal Length",
     ylab = "Sepal Width")

# Plot cluster centers
points(kc$centers[, c("Sepal.Length", "Sepal.Width")],
       col = 1:3,
       pch = 8,
       cex = 2)
